1 struct matrix {
 2     double ar[maxa][maxa];
 3     int n, m ; // n * m; 0 ~ n - 1, 0 ~ m - 1;
 4     matrix() {
 5         n = 4; //n
 6         m = 4; //m
 7         memset(ar, 0, sizeof(ar));
 8     }
 9     void clear() {
10         rep (i, n)
11             rep (j, m)
12                 ar[i][j] = 0;
13     }
14     void set_one() {
15         rep (i, n)
16             rep (j, m)
17                 ar[i][j] = 0;
18         rep (i, min(n, m))
19             ar[i][i] = 1;
20     }
21     void output() {
22         printf("%d %d\n", n, m);
23         rep(i, n) {
24             rep(j, m)
25                 printf("%.3f ", ar[i][j]);
26             printf("\n");
27         }
28         printf("\n");
29     }
30 };
31 matrix operator * (const matrix &a, const matrix &b) {
32     matrix c;
33     if(a.m != b.n) printf("a.m != b.n\n");
34     c.clear();
35     c.n = a.n;
36     c.m = b.m;
37     rep (i, a.n)
38         rep (j, b.m)
39         rep (k, a.m) {
40             c.ar[i][j] += a.ar[i][k] * b.ar[k][j];   //mod
41         }
42     return c;
43 }
