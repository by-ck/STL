 2 bool jiaodian(point a,point b,point c,point d,point &e)
 3 {
 4     double d1 = (b-a) * (c-a), d2 = (b-a) * (d-a),
 5            d3 = (d-c) * (a-c), d4 = (d-c) * (b-c);
 6     if(sgn(d1)*sgn(d2) > 0)
 7        return false;
 8     e = point( (c.x*d2 - d.x*d1) / (d2-d1) ,
 9                (c.y*d2 - d.y*d1) / (d2-d1) );
10     return true;
11 }
